试想一下如果不用不用redux-thunk怎么发起一个异步请求?
```js
const thunk = function ({dispatch, getState}) {
  return function (next) {
    return function (action) {
      if (typeof action === 'function') {
        return action(dispatch, getState)
      } else {
        return next(action)
      }
    }
  }
}
```
